package cn.com.ultrapower.topology
{
    import mx.binding.utils.BindingUtils;
    import mx.core.UIComponent;
    
    import flash.display.Graphics;
    
    public class Line extends UIComponent
    {
        private var _id:uint;
        // 节点坐标点
        private var _x1:Number;
        private var _y1:Number;
        private var _x2:Number;
        private var _y2:Number;
        
        // 起始节点
        private var _fromNode:Node = null;
        private var _toNode:Node   = null;
        
        // 属性
        private var _color:uint;
        
        public function Line(id:uint, data:XML){
            trace("连线", id, "创建成功!");
            _id = id;
            
            var tColor:String = data.@color;
            _color = parseInt(tColor.replace(/#/g, ''), 16);
        }
        
        /**
         * 绑定节点
         * */
        public function bindNodes(fromNode:Node, toNode:Node):void{
            bindFromNode(fromNode);
            bindToNode(toNode);
        }
        
        /**
         * 绑定始节点
         * */
        public function bindFromNode(node:Node):void{
            if(node){
                _fromNode = node;
                BindingUtils.bindProperty(this, "x1", node, {name: "x", getter: function():Number{return node.cx;}});
                BindingUtils.bindProperty(this, "y1", node, {name: "y", getter: function():Number{return node.cy;}});
                trace("始节点绑定成功:", node.Name);
            }else{
                trace("始节点绑定失败!");
            }
        }
        
        /**
         * 绑定终节点
         * */
        public function bindToNode(node:Node):void{
            if(node){
                _toNode = node;
                BindingUtils.bindProperty(this, "x2", node, {name: "x", getter: function():Number{return node.cx;}});
                BindingUtils.bindProperty(this, "y2", node, {name: "y", getter: function():Number{return node.cy;}});
                trace("终节点绑定成功:", node.Name);
            }else{
                trace("终节点绑定失败!");
            }
        }
        
        private function refresh():void{
            trace("重绘连线", _id);
            this.graphics.clear();
            this.graphics.lineStyle(1, _color);
            this.graphics.moveTo(_x1, _y1);
            this.graphics.lineTo(_x2, _y2);
        }
        
        public function get Id():uint{
            return _id;
        }
        
        public function set x1(n:Number):void{
            trace(_id, "x1:", n);
            _x1 = n;
            refresh();
        }
        
        public function set y1(n:Number):void{
            trace(_id, "y1:", n);
            _y1 = n;
            refresh();
        }
        
        public function set x2(n:Number):void{
            trace(_id, "x2:", n);
            _x2 = n;
            refresh();
        }
        
        public function set y2(n:Number):void{
            trace(_id, "y2:", n);
            _y2 = n;
            refresh();
        }

    }
}
